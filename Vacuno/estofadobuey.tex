\subsection{Estofado de buey}
\title{Estofado de buey}
\dline
\leftbgpic{Vacuno/estofadobuey1b}
\begin{info}
Raciones: 6 personas \\
Tiempo: 2h 30min
\end{info}
\\
\recipesection{Ingredientes}
\dline
\begin{ingredients}
 \item{1,5 Kg de Morcillo de buey}
 \item{1 cebolla grande}
 \item{2 ajos}
 \item{1 pimiento}
 \item{2 zanahorias}
 \item{2 tomates}
 \item{1 botella de vino tinto}
 \item{Sal y pimienta}
 \item{Laurel}
 \item{Agua}
 \item{2 patatas medianas (opcional)}
\end{ingredients}

\recipesection{Realización}
\dline
\step{1.}{
Corta las verduras y ponlas a pochar en aceite e incorporar el tomate rallado.
}
\dline
\step{2.}{
Corta la carne en trozos  e incorpórala a las
 verduras, rehoga durante 10 minutos y cúbrela
 con el vino tinto.
 }
 \dline
 \step{3.}{
 Dejar cocer lentamente
 durante 2 horas aproximadamente, añadiendo
 agua si lo necesita. Comprueba que la carne este tierna
 y si es necesario cuece un poco más sin dejar
 que se seque.
 }
 \dline
 \step{4.}{Añadir las patatas cortadas en cuadrados
 (opcional) y deja cocer 1/4 de hora más y servir.
}
