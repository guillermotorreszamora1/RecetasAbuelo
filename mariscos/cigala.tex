\subsection{Cigalas con ajo, perejil, aceite y sal gorda}
\title{Cigalas con ajo, perejil, aceite y sal gorda}
\leftbgpic{mariscos/cigala}

\begin{info}
.
\end{info}

\recipesection{Ingredientes}
\begin{ingredients}
\item{Cigalas}
\item{Ajo}
\item{Perejil}
\item{Aceite}
\item{Sal gorda}
\end{ingredients}
\recipesection{Realizacion}

\step{1.}{
Se abren las cigalas, se espolvorean el  ajo picado,
el perejil picado, un chorro de aceite y sal.
Se coloca en una bandeja de horno con la parte abierta
hacia arriba y se meten en el horno a media altura con el
grill hasta que se doren.
}
