\subsection{Garbanzos con sepia}
\title{Garbanzos con sepia}
\leftbgpic{legumbres/garbanzos_con_sepia.jpg}
\dline
\begin{info}
%9:20
 Tiempo de realización: 1h 30m \\
 Raciones: 4-5 personas
\end{info}
\recipesection{Ingedientes}
\dline
\begin{ingredients}
\item{1 cebolla}
\item{2 dientes de ajo}
\item{1 puerro}
\item{1 pimiento}
\item{1 tomate}
\item{1 cucharada de tomate frito}
\item{1 vaso de vino blanco}
\item{6-8 puñados de garbanzos}
\item{400g de sepia aprox.}
\end{ingredients}

\recipesection{Realización}
\dline
\step{1.}{
Poner a remojo la noche anterior los garbanzos
}
\step{2.}{
Hacer un sofrito en la olla con la cebolla, el ajo, el puerro, el pimiento y
el tomate.
}
\step{3.}{
Añadir una cucharada de tomate frito.
}
\step{4.}{
Añadir la sepia troceada y sofreir un poco.
}
\step{5.}{
Añadir un poco de vino blanco y dejar evaporar un par de minutos.
}
\step{6.}{
Añadir los garbanzos escurridos a la olla y
cubrir de agua unos 2 dedos aproximadamente.
}
\step{7.}{
Rectificar de sal, cerrar la olla y dejar cocer 50 minutos.
}
