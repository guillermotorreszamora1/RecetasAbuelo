\subsection{Chuletas de cerdo a la riojana}
\title{Chuletas de cerdo a la riojana}
\dline
\leftbgpic{Cerdo/ChuletasRiojana}
\begin{info}
 Raciones: 4 personas \\
 Tiempo de elaboración: 1h30
\end{info}
\recipesection{Ingredientes}
\dline
\begin{ingredients}
\item{4 Chuletas de cerdo}
\item{1 cebolla grande}
\item{2 pimientos verdes}
\item{1 pimiento rojo}
\item{1 vaso de tomate}
\item{\nicefrac{1}{2} vaso de vino blanco}
\item{1 chorro de aceite}
\item{Sal}
\item{Pimienta}
\item{(Opcional) 3 ó 4 pimientos de piquillo}
\end{ingredients}

\recipesection{Realización}
\dline
\step{1.}{
Pochar la cebolla junto con los pimientos cortados en juliana.
}
\step{2.}{
Añadir \nicefrac{1}{2} vaso de vino blanco, dejar evaporar unos minutos y añadir el tomate.
}
\step{3.}{
Dejar cocer aproximadamente 20 minutos.
}
\step{4.}{
En una sartén sellar las chuletas de cerdo previamente salpimentadas y añadirlas a la riojana. Dejar cocer 5 ó 10 minutos.
}
\step{5.}{
Se le puede añadir pimientos de piquillo cortados en tiras
}
