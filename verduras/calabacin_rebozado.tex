\subsection{Calabacín rebozado}
\title{Calabacín rebozado}
\leftbgpic{verduras/Calabacin_rebozado.jpg}
\dline
\begin{info}
 Tiempo de realización: 20m \\
 Raciones: 4 personass
\end{info}
\recipesection{Ingedientes}
\dline
\begin{ingredients}
 \item{1-2 Calabacines}
 \item{Sal}
 \item{Harina para rebozar}
 \item{2 huevos}
 \item{Aceite}
\end{ingredients}

\recipesection{Realización}
\dline
\step{1.}{
Cortar el Calabacín en rodajas. Añadirle sal y dejar escurir unos minutos en el colador para que echen el exceso de agua.
}
\step{2.}{
Pasar primero con harina y luego por huevo batido y freir. Dejarlos soltar el aceite en un plato con un papel absorvente.
}
