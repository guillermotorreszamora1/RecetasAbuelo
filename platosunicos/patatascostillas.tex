\subsection{Patatas con costillas}
\title{Patatas con costillas}
\dline
\begin{info}
 Raciones: 5 personas \\
 Tiempo de preparación: 1 hora
\end{info}
\recipesection{Ingredientes}
\dline
\begin{ingredients}
\item{1 cebolla}
\item{1 pimiento verde}
\item{2 ajos}
\item{4 Patatas}
\item{1 costillar de cerdo adobado partido}
\item{Perejil}
\item{Vino blanco}
\item{Aceite}
\end{ingredients}
\recipesection{Realización}
\dline
\step{1.}{
Pochar la cebolla, el pimiento y el ajo partido en trocitos.
}
\step{2.}{
Añadimos las costillas y las rehogamos.
}
\step{3.}{
Añadimos un majado de ajo, perejil y vino blanco.Dejamos evaporar un poco y añadimos las patatas cortadas.
}
\step{4.}{
Cubrimos de agua y dejamos cocer hasta que las costillas esten tiernas (aproximadamente 40 minutos/1 hora).
}
