\subsection{Conejo con salsa de tomate}
\title{Conejo con salsa de tomate}
\leftbgpic{Caza/conejotomate}
\begin{info}
Raciones: 6 personas \\
\end{info}
\recipesection{Ingredientes}
\begin{ingredients}
 \item{1 conejo troceado}
 \item{2 ajos}
 \item{Sal y pimienta}
 \item{1 cebola grande}
 \item{1,5 kg de tomate}
 \item{Orégano}
 \item{1 vasito de vino blanco}
 \item{1 cucharadita de Azúcar}
 \item{Aceite}
\end{ingredients}

\recipesection{Realización}

\step{1.}{
Freír el conejo con sal y pimienta y reservar
Para la salsa de tomate , picar la cebolla
 finita ponerla a pochar agregar los tomates
 partidos y pelados , la sal y el azúcar ,
 agregar el orégano , cuando este 1 hora
 aproximadamente agregar el vino y dejar
 ½ hora más , pasarlo por el pasapuré e
 incorporar conejo , dejarlo 10 minutos más y servir
}
