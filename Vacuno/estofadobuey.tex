\subsection{Estofado de buey}
\title{Estofado de buey}
\begin{info}
Raciones: 6 personas \\
Tiempo: ?
\end{info}
\recipesection{Ingredientes}
\begin{ingredients}
 \item{1,5 Kg de Morcillo de buey}
 \item{1 cebolla grande}
 \item{2 ajos}
 \item{1 pimiento}
 \item{2 zanahorias}
 \item{2 tomates}
 \item{1 botella de vino tinto}
 \item{Sal y pimienta}
 \item{Laurel}
 \item{Agua}
 \item{0.3 kg de patatas (opcional)}
\end{ingredients}
\recipesection{Realización}
\step{1.}{
Corta las verduras y ponlas a pochar en aceite.
}
\step{2.}{
Corta la carne en trozos  e incorpórala a las
 verduras , rehoga durante 10 minutos y cúbrela
 con el vino tinto.
 }
 \step{3.}{
 Ponlo a cocer lentamente
 durante 2 horas aproximadamente , añadiendo
 agua si lo necesita. Comprueba que este tierna
 y si es necesario cuece un poco más sin dejar
 que se seque. Añade 0,5 kilos de patatas
 (opcional) y deja cocer 1/4 de hora más y servir.
}
