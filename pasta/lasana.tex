\subsection{Lasaña de Jamón de York con Queso gratinada con Parmesano}
\title{Lasaña de Jamón de York con Queso gratinada con Parmesano}
\leftbgpic{pasta/lasana}
\begin{info}
 Raciones: 8 personas
\end{info}
\recipesection{Ingredientes}
\begin{ingredients}
 \item{Salsa de Tomate}
 \item{16 Lonchas de Jamón de York}
 \item{16 Lonchas de Queso}
 \item{1L de Bechamel}
 \item{24 Láminas de canelones}
\end{ingredients}
\recipesection{Realización}

\step{1.}{ Cocemos la pasta, ponemos una capa de tomate
en la bandeja del horno, después ponemos una
capa de Láminas de pasta, después una capa de
lonchas de queso, otra capa de pasta, una capa
de tomate, otra capa de jamón , se cubre todo
con bechamel y se le echa queso parmesano, se
mete en el horno durante aproximadamente 20
minutos a 190ºC}
