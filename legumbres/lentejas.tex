\subsection{Lentejas con chorizo}
\title{Lentejas con chorizo}
\leftbgpic{legumbres/lentejas}
\begin{info}
 Tiempo de realización 1,5 horas \\
 Raciones; 6 personas
\end{info}
\recipesection{Ingedientes}
\begin{ingredients}
 \item{400g de lentejas}
 \item{\nicefrac{1}{2} cebolla}
 \item{1 zanahoria}
 \item{1 pimiento verde}
 \item{1 tomate}
 \item{1 hoja de laurel}
 \item{1 cabeza de ajo}
 \item{1 cucharadita de pimentón}
 \item{1 chorito de aceite}
 \item{200 g de chorizo (opcional)}
\end{ingredients}
\recipesection{Realización}
\step{1.}{
Se pone todo en una cazuela (excepto las patatas) en crudo,
se pone a cocer a fuego lento hasta que estén tiernas(1,5 horas
aproximadamente)añadiendole agua fría siempre que haga falta,
.Si se echan patatas, se echan cortadas en daditos pequeños 20
minutos antes de terminar la cocción.Si quedan caldosas se le puede
añadir una cucharadita de pan rallado
}
