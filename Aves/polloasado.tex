\subsection{Pollo asado}
\title{Pollo asado}
\leftbgpic{Aves/polloasado}
\dline
\begin{info}
Raciones: 6 personas \\
Tiempo de preparación: 1h45 min
\end{info}
\recipesection{Ingredientes}
\dline
\begin{ingredients}
\item{6 cuartos traseros de pollo}
\item{Sal y pimienta}
\item{Aceite}
\item{1 vaso de vino blanco seco}
\item{\nicefrac{1}{2} vaso de agua}
\item{3 patatas (opcional)}
\end{ingredients}

\recipesection{Realización}
\dline
\step{1.}{
Precalentar el horno a 200ºC
}
\newline
\step{2.}{
Salpimentar el pollo y cortar las patatas en rodajas. Pasarlo todo a una fuente con la patatas de base y el pollo con la piel hacia abajo.
}
\newline
\step{3.}{
Agregar el vino, el agua y un chorro de aceite sobre el pollo. 
}
\newline
\step{4.}{
Meterlo en el horno durante \nicefrac{3}{4} de hora. Darle la vuelta, regarlo con el caldo y meterlo otros \nicefrac{3}{4} de hora bajando la temperatura a 180ºC.
}
\newline
\step{5.}{
Sacarlo del horno y servir.
}
