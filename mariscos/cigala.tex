\subsection{Cigalas con ajo, perejil, aceite y sal gorda}
\title{Cigalas con ajo, perejil, aceite y sal gorda}
\leftbgpic{mariscos/cigala}
\dline
\begin{info}
.
\end{info}

\recipesection{Ingredientes}
\dline
\begin{ingredients}
\item{Cigalas}
\item{Ajo}
\item{Perejil}
\item{Aceite}
\item{Sal gorda}
\end{ingredients}

\recipesection{Realizacion}
\dline
\step{1.}{
Se abren las cigalas con una tijerita y se espolvorean con el ajo y prejil
picado, un chorro de aceite y sal.
}
\step{2.}{
Se colocan en una bandeja de horno con la parte abierta
hacia arriba y se meten en el horno a media altura con el
grill puesto hasta que se doren los ajos.
}
