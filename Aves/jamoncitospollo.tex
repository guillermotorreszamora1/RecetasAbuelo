\subsection{Jamoncitos de pollo encebollados}
\title{Jamoncitos de pollo encebollados}
\leftbgpic{Aves/jamoncitospollo}
\dline
\begin{info}
Raciones: 6 personas \\
Tiempo de preparación: 1h30 min
\end{info}
\recipesection{Ingredientes}
\dline
\begin{ingredients}
\item{18 muslos de pollo}
\item{2 cebollas}
\item{2 dientes de ajo}
\item{Sal}
\item{Pimienta}
\item{Aceite}
\item{1 vaso de vino blanco}
\item{1 vaso de agua}
\item{2 cucharadas de vinagre}
\end{ingredients}
\recipesection{Realización}
\dline
\step{1.}{
Salpimentar los muslos de pollo y dorar en una cazuella con aceite. Retirar y reservar
}
\step{2.}{
En ese aceite, pochar la cebolla cortada en juliana y el ajo.
}
\step{3.}{
Añadir los muslos de pollo, el vaso de vino blanco, el agua y las 2 cucharadas de vinagre.
Dejar cocer a fuego lento, aproximadamente entre 45 minutos/1 hora.
Mover en vaivén de vez en cuando.
}
