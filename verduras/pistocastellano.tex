\subsection{Pisto Castellano}
\title{Pisto Castellano}
\leftbgpic{verduras/pistocastellano}
\begin{info}
 Raciones: ?? personas
\end{info}
\recipesection{Ingredientes}
\begin{ingredients}
 \item{1 kg de cebolla}
 \item{1.5 kilos de calabacín}
 \item{\nicefrac{1}/{2} kg de pimientos}
 \item{3 ajos}
 \item{Perejil}
 \item{Aceite}
 \item{Sal}
\end{ingredients}
\recipesection{Realización}
\step{1.}{
Pica la cebolla menuda y ponla en una sartén
a freír con el pimiento cuando este un poco blanda
añadir los ajos y el calabacín cortado en
cuadritos pequeños, agrega el tomate también
 cortado pequeño, el perejil picado. Deja cocer
 tapado a fuego lento hasta que este tierno.
 Y servir
}
