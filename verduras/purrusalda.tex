\subsection{Purrusalda}
\title{Purrusalda}
\leftbgpic{verduras/Purrusalda.jpg}
\dline
\begin{info}
 Raciones: 4-5 personas \
%10:12
 Tiempo de preparación: 1 hora 15 minutos
\end{info}
\recipesection{Ingredientes}
\dline
\begin{ingredients}
\item{4 puerros}
\item{1 cebolleta}
\item{1 o 2 zanahorias}
\item{4 patatas}
\item{1 litro de caldo de pollo}
\item{2 ajos}
\item{lomo de bacalao}
\item{1 cayena}
\end{ingredients}
\recipesection{Realizacion}
\dline
\step{1.}{
Pochar puerro, cebolla y zanahoria.
}
\newline
\step{2.}{
Añadir las patatas en rodajas o tacos.
}
\newline
\step{3.}{
Cubrir con el caldo y cocer 15 minutos
}
\newline
\step{4.}{
Confitar el bacalao en una sarten. Retirar y hacer lastas.
En ese aceite freir 2 ajos laminados junto con una cayena.
}
\newline
\step{5.}{
Añadirlo a la cazuela y dejar cocer 10 minutos.
}
