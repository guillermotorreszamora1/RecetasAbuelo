\subsection{Arroz con bacalao}
\title{Arroz con bacalao}
\leftbgpic{platosunicos/arroz_bacalao}
\dline
\begin{info}
 Raciones: 4 ó 5 personas \\
%19:50
 Tiempo de preparación: ?? horas
\end{info}
\recipesection{Ingredientes}
\dline
\begin{ingredients}
\item{1 cebolla}
\item{1 pimiento verde}
\item{1 pimiento rojo}
\item{2 ajos}
\item{4 patatas}
\item{4 puñados de arroz}
\item{1 hoja de laurel}
\item{Perejil}
\item{Pimentón}
\item{4 rodajas de bacalao}
\end{ingredients}

\recipesection{Realización}
\dline
\step{1.}{
Pochamos la cebolla y los pimientos en juliana junto con el ajo.
}
\step{2.}{
Añadimos las patatas cortadas en medallones y una cucharada de pimentón.
Cubrimos con agua y dejamos cocer 15 minutos.
}
\step{3.}{
Añadimos el arroz y el bacalao y un majado de ajo, perijil y vino blanco
y dejamos cocer aproximadamente 20 minutos.
}
