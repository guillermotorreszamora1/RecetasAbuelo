\subsection{Ensalada de pasta}
\title{Ensalada de pasta}
\leftbgpic{pasta/ensaladapasta}
\begin{info}
Raciones: 6 personas \\
\end{info}
\recipesection{Ingredientes}
\begin{ingredients}
 \item{\nicefrac{1}{2} kg de pasta (lacitos o espirales) }
 \item{300g de jamón de york en dados}
 \item{4 huevos cocidos picados}
 \item{1 Tomate fresco picado}
 \item{1 Pimiento morrón picado}
 \item{\nicefrac{1}{4} de bonito en aceite}
 \item{2 manzanas en daditos pequeños}
 \item{100g de nueces partidas en trocitos}
 \item{300g de salsa rosa (o salsa coctel)}
\end{ingredients}
\recipesection{Realización}

\step{1.}{
Cuece la pasta en abundante agua con sal
(Tiempo que marque el fabricante), enfriar,
mezclas la pasta con los el resto de los ingredientes,
mezclar bien y enfriar en el frigorífico, puedes añadir
 pasas de corinto.
}
