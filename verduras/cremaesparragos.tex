\subsection{Crema de esparragos}
\title{Crema de esparragos}
\dline
\begin{info}
 Tiempo de realización: 40m \\
 Raciones: 5 personass
\end{info}
\recipesection{Ingredientes}
\dline
\begin{ingredients}
\item{2 botes de esparragos blancos}
\item{3 patatas}
\item{1 pastilla de caldo de pollo}
\item{1 brick de nata de cocinar}
\end{ingredients}

\recipesection{Realización}
\dline
\step{1.}{
 Quitar el liquido de los esparragos y picarlos.
 Pelar las patatas y picarlas también y echar todo en una olla
}
\dline
\step{2.}{
Poner a hervir aproximadamente 20 minutos, cuando rompa a hervir echar la pastilla
de caldo.
}
\dline
\step{3.}{
Echar el brick de nata de cocinar y triturar.
}
