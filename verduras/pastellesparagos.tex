\subsection{Pastel de esparragos}
\title{Pastel de esparragos}
\leftbgpic{verduras/pastel_esparragos}
\dline
\begin{info}
 Tiempo de realización: 1 hora \\
 Raciones: 5 personas
\end{info}
\recipesection{Ingredientes}
\dline
\begin{ingredients}
\item{3 huevos}
\item{4 rebanadas de pan bimbo}
\item{1 bote de esparragos blancos}
\item{1 tetabrik de nata}
\item{Mantequilla para untar el molde}
\end{ingredients}
\recipesection{Realización}
\dline
\step{1.}{
Batir todos los ingredientes y ponerlos en un molde untado con mantequilla.
}
\step{2.}{
Meterlo en el horno al baño maria durante 40 minutos a 180º
}
