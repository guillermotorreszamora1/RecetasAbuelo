\subsection{Atún con tomate}
\title{Atún con tomate}
\dline
\leftbgpic{pescados/atun_tomate_1}
\begin{info}
Raciones: 6 personas \\
Tiempo de preparación: 40 minutos
\end{info}
\recipesection{Ingredientes}
\dline
\begin{ingredients}
\item{800g de bonito}
\item{1kg de tomate natural triturado}
\item{\nicefrac{1}{2} pimiento verde}
\item{\nicefrac{1}{2} pimiento rojo}
\item{1 cebolla}
\item{1 ajo}
\item{Aceite}
\item{Sal}
\end{ingredients}

\recipesection{Realización}
\dline
\step{1.}{
Vamos a realizar una salsa de tomate.
Pochamos la cebolla, el ajo y los pimientos.
}
\dline
\step{2.}{
Añadimos el tomate natural triturado y lo dejamos sofreir durante 40 minutos aproximadamente.
Rectificamos de sal.
}
\dline
\step{3.}{
Añadimos al tomate frito los trozos de atún previamente salpimientados, dejamos cocer a fuego lento
5 ó 6 minutos. Se pude acompañar este placo con arroz blanco.
}
