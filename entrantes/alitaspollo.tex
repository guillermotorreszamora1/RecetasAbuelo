\subsection{Alitas de pollo al ajillo}
\title{Alitas de pollo al ajillo}
\dline
\leftbgpic{entrantes/alitaspollo}
\begin{info}
 Raciones: x personas
\end{info}
\recipesection{Ingredientes}
\dline
\begin{ingredients}
\item{10 alitas de pollo}
\item{Aceite}
\item{Sal y pimienta}
\item{4 dientes de ajo laminados}
\item{1 vasito de vino blanco}
\item{2 cucharadas de vinagre}
\end{ingredients}

\recipesection{Realización}
\dline
\step{1.}{
Freír las alas hasta que estén
doradas y reservar.}
\dline
\step{2.}{
En ese aceite se fríen los ajos y cuando empiecen a
dorarse se aparta la sartén del fuego.}
\dline
\step{3.}{
Agregamos
el vino con el vinagre y se vuelve a poner
en el fuego e incorporamos las alitas. Dejamos hervir tres minutos más.
}
