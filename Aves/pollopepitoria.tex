\subsection{Pollo en pepitoria}
\title{Pollo en pepitoria}
\dline
\leftbgpic{Aves/pollopepitoria}
\begin{info}
Raciones: 4 ó 5 personas \\
Tiempo de preparación: 1 hora
\end{info}
\recipesection{Ingredientes}
\dline
\begin{ingredients}
\item{1 kg de pollo troceado}
\item{1 Cebolla}
\item{4 dientes de ajo}
\item{Perejil}
\item{12 ó 14 almendras}
\item{Vino blanco}
\item{1 huevo cocido}
\item{4 ó 5 cucharadas de harina}
\item{1 pastilla de caldo de pollo}
\end{ingredients}

\recipesection{Realización}
\dline
\step{1.}{
Salpimentar el pollo, enharinar y sellarlo en una sarten y reservar.
}
\step{2.}{
En el mismo aceite, pochar la cebolla y 3 dientes de ajo. Una vez pochado, añadir una cucharadita de
harina, tostar un poquito y añadir un majado con las almendras, 1 ajo, vino blanco y perejil. Dejar evaporar
un poco el vino, añadir el pollo y cubrir con agua y una pastilla de caldo de pollo.
}
\step{3.}{
Añadir el huevo picado y dejar cocer aproximadamente 3/4 de hora a fuego lento.
}
