\subsection{Espaguetis con salsa boloñesa y queso mozzarella}
\title{Espaguetis con salsa boloñesa y queso mozzarella}
\leftbgpic{pasta/espaguetis}
\begin{info}
Raciones: 5 personas
\end{info}

\recipesection{Ingredientes}
\begin{ingredients}
 \item{\nicefrac{1}{2} kg de espaguetis}
 \item{\nicefrac{1}{2} kg de carne picada}
 \item{1 cebolla}
 \item{2 ajos}
 \item{6 tomates maduros}
 \item{1 vasito de vino blanco}
 \item{Albaca}
 \item{100 g de queso mozzarella}
\end{ingredients}
\recipesection{Realización}
\step{1.}{
Cuece la pasta según fabricante , para la salsa
pica la cebolla menuda con los ajos , pochala en
una sartén , cuando este  pochado agrega los
tomates pelados , en trozos muy pequeño y la
carne , dale una vueltas , agrega el vino blanco
y la albaca , dale unas vueltas, tápalo y cuece
lentamente durante 1,5 horas removiendo despacio
 .Sirve los espaguetis con la salsa y la mozzarella
}
