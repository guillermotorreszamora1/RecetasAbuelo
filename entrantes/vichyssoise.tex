\subsection{Vichyssoise}
\title{Vichyssoise}
\leftbgpic{entrantes/Vichyssoise2}
\begin{info}
 Raciones: 10 personas \\
 Tiempo de preparación: 1 hora
\end{info}
\recipesection{Ingredientes}
\begin{ingredients}
 \item{1 cebolla grande}
 \item{4 puerros (solo la parte blanca)}
 \item{3 patatas gordas (aproximadamente 1/2 kilo)}
 \item{1 cucharada sopera de Mantequilla}
 \item{1 litro de Caldo}
\item{1 litro de leche (aproximadamente)}
\item{200g de nata para cocinar}
\end{ingredients}
\recipesection{Realización}

\step{1.}{
En una cacerola poner la mantequilla a deretir y pochar la
cebolla y el puerro.
}
\step{2.}{
Añadir las patatas cortadas, dar un par de vueltas y añadir el caldo
(Puede ser agua y una pastilla de avecren en su defecto).
}

\step{3.}{
Dejar cocer 1/2 hora aproximadamente, pasar la turmix y añadir la leche.
}

\step{4.}{
Por último añadirle la nata de cocinar, batir y dejar enfriar. En caso de quedar algún
grumo pasarlo por un colador o chino.
}
