\subsection{Carrilleras de Cerdo/Ternera}
\title{Carrilleras de Cerdo/Ternera}
\leftbgpic{Cerdo/Carrilleras.jpg}
\dline
\begin{info}
 Raciones: 6 personas \\
 Tiempo de preparación: 1 hora
\end{info}
\recipesection{Ingredientes}
\dline
\begin{ingredients}
\item{1,7-2kg de carrilleras}
\item{1 cebolla}
\item{1 zanahoria}
\item{3 ajos}
\item{ \nicefrac{1}{2} pimiento rojo}
\item{Sal}
\item{Pimienta}
\item{1 vaso de agua}
\item{1 vaso de vino tinto}
\item{1 clavo}
\item{Harina}
\end{ingredients}

\recipesection{Realización}
\dline
\step{1.}{
Enharinar la carne previamente salpimentada y sellar en una sartén. Retirar
}
\step{2.}{
En ese aceite pochar las verduras(cebolla, zanahoria, ajos y el pimiento). Una vez pochado añadir el agua y el vino.
Dejar evaporar un poco y añadir la carne.
}
\step{3.}{
Cerrar la olla 40 minutos.
}
\newline
\step{4.}{
Filetear las carrilleras y pasar las verduras por la turmix
}
