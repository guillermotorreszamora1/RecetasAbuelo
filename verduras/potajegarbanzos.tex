\subsection{Potaje de garbanzos}
\title{Potaje de garbanzos}
\dline
\begin{info}
Raciones: 4 ó 5 personas
\end{info}
\recipesection{Ingredientes}
\dline
\begin{ingredients}
\item{1 cebolla}
\item{1 ajo}
\item{1 tomate rallado}
\item{6 puñados de garbanzos}
\item{2 huevos duros}
\item{1/2 kilo de espinacas}
\item{1 cucharadita de pimentón}
\end{ingredients}

\recipesection{Realización}
\dline
\step{1.}{
Pochar la cebolla, el ajo y una vez pochado añadir el tomate rallado y una cucharadita de pimentón.
Rehogar 10 minutos.
}
\step{2.}{
Añadir los garbanzos (previamente remojados desde la noche anterior) y cubrir con agua.
Añadir la sal y dejar cocer en olla a presión durante 40 minutos.
}
\step{3.}{
Abrir la olla, añadir las espinacas y los 2 huevos cocidos picados y dejar cocer unos 5 minutos
}
