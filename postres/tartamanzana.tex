\subsection{Tarta de Manzana}
\title{Tarta de Manzana}
\leftbgpic{postres/tartamanzana}
\begin{info}
Raciones: 4 personas
\end{info}
\recipesection{Ingredientes}
\begin{ingredients}
 \item{6 huevos}
 \item{250g de harina}
 \item{100g de mantequilla}
 \item{150g de azúcar}
 \item{1 sobrecito de levadura}
 \item{1 limón}
 \item{3 manzanas}
%  \item{3 cucharadas de azúcar glas}
\end{ingredients}
\recipesection{Realización}
\step{1.}{
Batir los huevos con el azúcar
 hasta obtener una pasta suave
 y esponjosa. Añadir poco a poco
 la ralladura, el zumo de limón,
 y la mantequilla derretida, incorporar
 la levadura y mezclar todo bien.
}
\step{2.}{
 Pelar las manzanas y cortarlas en cuñas finas.
 Colocarlas en una tartera engrasada con
 mantequilla y enharinada.
 }
 \step{3.}{
 Verter encima la masa e introducir en el horno precalentado
 a 180ºC, durante 30 minutos. Terminada la
 cocción, desmoldar la tarta sobre una fuente
 de servir. Espolvorear la tarta con el azúcar
 glas y servir, templada o fría.
}
