\subsection{Almejas a la marinera}
\title{Almejas a la marinera}
\leftbgpic{entrantes/almejas_marinera2.jpg}
\dline
\begin{info}
 Raciones: 4 personas \\
 Tiempo de preparacion: 15 minutos
\end{info}
\recipesection{Ingredientes}
\dline
\begin{ingredients}
 \item{1 malla de almejas}
 \item{2 ajos}
 \item{ \nicefrac{1}{2} cebolla}
 \item{Perejil}
 \item{1 vaso de vino blanco}
 \item{1 cucharada de harina}
 \item{Sal}
\end{ingredients}
\recipesection{Realización}
\dline
\step{1.}{
Pochar la media cebolla y el ajo.
}
\step{2.}{
Añadir una cucharada de harina, rehogar un poquito y añadir el majado(Ajo, perejil y vino blanco machacados)
}
\step{3.}{
Añadir las almejas(Nota: las almejas ponerlas un rato en agua con un puñadito de sal para que echen la tierra y escurrir)
y un poco de agua. Remover un poco hasta que se abran las almejas.
}
