\subsection{Sopa Repollo}
\title{Sopa repollo}
\dline

\begin{info}
Tiempo de realización: 45 minutos
Raciones: 5 ó 6 personas
\end{info}
\recipesection{Ingredientes}
\dline
\begin{ingredients}
\item{ 1/2 repollo}
\item{2 patatas}
\item{3 dientes de ajo}
\item{1 cucharadita de pimentón}
\item{1 litro de caldo de pollo}
 \item{Sal}
\end{ingredients}

\recipesection{Realización}
\dline
\step{1.}{
Picar el medio repollo en cuadraditos y las 2 patatas. Echarlos en una olla con el litro de caldo y dejar hervir, aproximadamente 1/2 hora.
}
\dline
\step{2.}{
Sofreir los dientes de ajos en laminas y añadir la cucharadita de pimentón con el fuego a menos temperatura.
}
\dline
\step{3.}{
Añadir la ajada al guiso y dejar que se integre durante 2 minutos a fuego lento.
}
