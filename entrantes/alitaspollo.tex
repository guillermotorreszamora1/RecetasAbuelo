\subsection{Alitas de pollo al ajillo}
\title{Alitas de pollo al ajillo}
\leftbgpic{entrantes/alitaspollo}
\begin{info}
 Raciones: x personas
\end{info}
\recipesection{Ingredientes}
\begin{ingredients}
\item{10 alitas de pollo}
\item{Aceite}
\item{Sal y pimienta}
\item{4 dientes de ajo laminados}
\item{1 vasito de ajo laminados. Vino blanco?}
\end{ingredients}
\recipesection{Realización}

\step{1.}{
Freír las alas hasta que estén
doradas y reservar en ese aceite
se fríen los ajos cuando empiecen a
dorarse se aparta la sartén y se agrega
el vino con el vinagre se vuelve a poner
en el fuego se incorporan las alas. Y las
dejas dos o tres minutos más.
}
