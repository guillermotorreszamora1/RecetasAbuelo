
\subsection{Lasaña de Jamón de York con Queso gratinada con Parmesano}
\title{Lasaña de Jamón de York con Queso gratinada con Parmesano}
\leftbgpic{pasta/lasana_jamon1}
\dline
\begin{info}
 Raciones: 8 personas
\end{info}
\recipesection{Ingredientes}
\dline
\begin{ingredients}
 \item{Salsa de Tomate}
 \item{16 Lonchas de Jamón de York}
 \item{16 Lonchas de Queso}
 \item{1L de Bechamel}
 \item{24 Láminas de canelones}
\end{ingredients}

\recipesection{Realización}
\dline
\step{1.}{
Cocemos la pasta según fabricante. Escurrimos y dejamos secar en un paño limpio
}
\dline
\step{2.}{
Ponemos una capa de tomate
en la bandeja del horno, después ponemos una
capa de láminas de pasta, después una capa de
lonchas de jamón y queso y untamos con tomate frito. Luego otra capa de pasta y volvemos a repetir el
paso anterior y así hasta acabar las laminas.
}
\dline
\step{3.}{
Cubrir con bechamel y queso parmesano. Se
mete en el horno durante aproximadamente 20
minutos a 190ºC hasta que este dorado}
