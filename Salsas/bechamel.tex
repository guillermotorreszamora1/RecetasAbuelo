\subsection{Salsa Bechamel}
\title{Salsa Bechamel}
\dline
\recipesection{Ingredientes}
\dline
\begin{ingredients}
\item{90g de harina}
\item{90g de mantequilla}
\item{900g de leche}
\item{Sal}
\item{Nuez moscada}
\end{ingredients}

\recipesection{Realización}
\dline
\step{1.}{
Pon la mantequilla en una cazuela, cuando esta derretida agregar la harina
 e ir removiendo con una cuchara de madera o con una varilla para que se dore un poco.
}
\dline
\step{2.}{Agregar la leche caliente poco a poco y sin dejar de
 remover con el fuego lento y mantener la cocción sin
 que hierva.
 }
 \dline
 \step{3.}{
 Añadir la sal y la nuez moscada y dejar cocer hasta que alcance la textura
 adecuada
}
