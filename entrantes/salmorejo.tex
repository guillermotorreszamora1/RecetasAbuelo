\subsection{Salmorejo}
\title{Salmorejo}
\dline
\leftbgpic{entrantes/salmorejo}
\begin{info}
 Raciones: 4 personas
\end{info}
\recipesection{Ingredientes}
\begin{ingredients}
\dline
\item{4 o 5 ajos}
\item{1,5 kg de tomates maduros muy rojos}
\item{\nicefrac{1}{2} barra de pan candeal}
\item{100ml de aceite}
\item{Sal}
\group{Guarnición}
\item{2 huevos cocidos picados finitos}
\item{Jamón serrano picado finito}
\end{ingredients}

\recipesection{Realización}
\dline
\step{1.}{
Emulsiona el aceite con los ajos usando la batidora.
Échalo a un cuenco
}
\step{2.}{Tritura los tomates, pásalos
 por el chino y júntalo con el aceite.
}
\step{3.}{
 Remoja el pan
 con agua, bátelo bien por la batidora y pásalo por chino.
 }
 \step{4.}{
 Incorporándolo todo a la cazuela, echa sal al gusto,
 bátelo bien y ponlo a enfriar.
 }
 \step{5.}{
La guarnición se la  sirve cada comensal en su taza.
}
